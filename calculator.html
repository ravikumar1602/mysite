            import tkinter as tk

            # === Button Click Function ===
            def button_click(value):
                current = entry.get()
                entry.delete(0, tk.END)
                entry.insert(0, current + value)
            
            def clear_entry():
                entry.delete(0, tk.END)
            
            def calculate():
                try:
                    result = eval(entry.get())
                    entry.delete(0, tk.END)
                    entry.insert(0, str(result))
                except:
                    entry.delete(0, tk.END)
                    entry.insert(0, "Error")
            
            # === Create Window ===
            root = tk.Tk()
            root.title("Classic Calculator")
            root.geometry("460x570")
            root.resizable(False, False)
            
            # === Fonts ===
            font_large = ("Helvetica", 24)
            font_button = ("Helvetica", 20, "bold")
            
            # === Entry Field ===
            entry = tk.Entry(root, font=font_large, bd=10, relief=tk.RIDGE, justify="right")
            entry.grid(row=0, column=0, columnspan=4, padx=10, pady=20, ipady=15)
            
            # === Button Layout ===
            buttons = [
                ('7', 1, 0), ('8', 1, 1), ('9', 1, 2), ('+', 1, 3),
                ('4', 2, 0), ('5', 2, 1), ('6', 2, 2), ('-', 2, 3),
                ('1', 3, 0), ('2', 3, 1), ('3', 3, 2), ('*', 3, 3),
                ('C', 4, 0), ('0', 4, 1), ('=', 4, 2), ('/', 4, 3),
            ]
            
            for (text, row, col) in buttons:
                if text == "=":
                    btn = tk.Button(root, text=text, font=font_button, bg="green", fg="white",
                                    command=calculate, width=5, height=2)
                elif text == "C":
                    btn = tk.Button(root, text=text, font=font_button, bg="red", fg="white",
                                    command=clear_entry, width=5, height=2)
                else:
                    btn = tk.Button(root, text=text, font=font_button,
                                    command=lambda t=text: button_click(t), width=5, height=2)
            
                btn.grid(row=row, column=col, padx=10, pady=10)
            
            # === Run App ===
            root.mainloop()
